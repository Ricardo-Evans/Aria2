020package com.inchok.aria2;
021
022import java.io.Serializable;
023import java.util.Date;
024import java.util.List;
025import java.util.Objects;
026
027/**
028 * The bit torrent meta info data retrieved from ".torrent" file.
029 * To get these data, call DownloadHandle.getBtMetaInfoData().
030 *
031 * @author inCHOK
032 * @version Version 1.0
033 * @see DownloadHandle#getBtMetaInfo()
034 */
035public class BtMetaInfoData implements Serializable {
036    private List<List<String>> announceList;
037    private String comment;
038    private Date creationDate;
039    private BtFileMode mode;
040    private String name;
041
042    BtMetaInfoData(List<List<String>> announceList, String comment, Date creationDate, BtFileMode mode, String name) {
043        this.announceList = announceList;
044        this.comment = comment;
045        this.creationDate = creationDate;
046        this.mode = mode;
047        this.name = name;
048    }
049
050    /**
051     * To get the announce URIs of the torrent file.
052     * If torrent contains <cite>announce</cite> instead of <cite>announce-list</cite>, <cite>announce</cite> is converted into <cite>announce-list</cite> format.
053     *
054     * @return Return the list of lists of announce URIs of the torrent file.
055     */
056    public List<List<String>> getAnnounceList() {
057        return this.announceList;
058    }
059
060    /**
061     * To get the comment of the torrent file.
062     * <cite>comment-utf8</cite> is used if available.
063     *
064     * @return Return the comment of the torrent file.
065     */
066    public String getComment() {
067        return this.comment;
068    }
069
070    /**
071     * To get the create time of the torrent file.
072     *
073     * @return Return the create time of the torrent file.
074     */
075    public Date getCreationDate() {
076        return this.creationDate;
077    }
078
079    /**
080     * To get the bit torrent file mode.
081     *
082     * @return Return the bit torrent file mode
083     * @see BtFileMode
084     */
085    public BtFileMode getMode() {
086        return this.mode;
087    }
088
089    /**
090     * To get the <cite>name</cite> in <cite>info</cite> dictionary.
091     * <cite>name-utf8</cite> is used instead of <cite>name</cite> if available.
092     *
093     * @return Return the <cite>name</cite> in <cite>info</cite> dictionary.
094     */
095    public String getName() {
096        return this.name;
097    }
098
099    /**
100     * To compare whether the two BtMetaInfoData is equal.
101     *
102     * @param o The object to be compared with.
103     * @return Return true if the two objects are equal, otherwise return false.
104     */
105    @Override
106    public boolean equals(Object o) {
107        if (this == o) return true;
108        if (!(o instanceof BtMetaInfoData)) return false;
109        BtMetaInfoData that = (BtMetaInfoData) o;
110        return Objects.equals(getAnnounceList(), that.getAnnounceList()) &&
111                Objects.equals(getComment(), that.getComment()) &&
112                Objects.equals(getCreationDate(), that.getCreationDate()) &&
113                getMode() == that.getMode() &&
114                Objects.equals(getName(), that.getName());
115    }
116
117    /**
118     * To get the hash code of the BtMetaInfoData.
119     *
120     * @return Return the hash code of the BtMetaInfoData.
121     */
122    @Override
123    public int hashCode() {
124        return Objects.hash(getAnnounceList(), getComment(), getCreationDate(), getMode(), getName());
125    }
126
127    /**
128     * To convert the BtMetaInfoData into a String.
129     *
130     * @return Return the String result.
131     */
132    @Override
133    public String toString() {
134        return "BtMetaInfoData{" +
135                "announceList=" + announceList +
136                ", comment='" + comment + '\'' +
137                ", creationDate=" + creationDate +
138                ", mode=" + mode +
139                ", name='" + name + '\'' +
140                '}';
141    }
142}
