022import java.util.List;

024/**
025 * The static class which operates the global operations.
026 * Before you do anything with Aria2, you must call Aria2.initialize() first to initialize. After you finish your task, you need to call Aria2.deInitialize() to release the static global resources.
027 *
028 * @author inCHOK
029 * @version Version 1.0
030 * @see Aria2#initialize()
031 * @see Aria2#deInitialize()
032 */
033public class Aria2 {
034    static {
035        System.loadLibrary("aria2-native");
036    }
037
038    /**
039     * The response code when any operation succeed.
040     */
041    public static final int RESPONSE_OK = 0;
042
043    private Aria2() {
044    }
045
046    /**
047     * Initialize the global data.
048     * Call this static method <strong>only once</strong> before any other operations of Aria2.
049     *
050     * @return Return Aria2.RESPONSE_OK if succeed, or any other negative error code.
051     * @see Aria2#RESPONSE_OK
052     */
053    public static int initialize() {
054        return Aria2.initializeNative();
055    }
056
057    /**
058     * Release all the global data.
059     * Call this static method <strong>only once</strong> after all the tasks of Aria2 are finished.
060     *
061     * @return Return Aria2.RESPONSE_OK if succeed, or any other negative error code.
062     * @see Aria2#RESPONSE_OK
063     */
064    public static int deInitialize() {
065        return Aria2.deInitializeNative();
066    }
067
