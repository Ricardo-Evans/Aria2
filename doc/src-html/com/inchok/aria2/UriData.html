020package com.inchok.aria2;
021
022import java.io.Serializable;
023import java.util.Objects;
024
025/**
026 * The uri data get from FileData.getUris().
027 *
028 * @author inCHOK
029 * @version Version 1.0
030 * @see FileData#getUris()
031 */
032public class UriData implements Serializable {
033    private String uri;
034    private UriStatus status;
035
036    UriData(String uri, UriStatus status) {
037        this.uri = uri;
038        this.status = status;
039    }
040
041    /**
042     * To get the uri.
043     *
044     * @return Return the uri.
045     */
046    public String getUri() {
047        return this.uri;
048    }
049
050    /**
051     * To get the status of the uri.
052     *
053     * @return Return the status of the uri.
054     * @see UriStatus
055     */
056    public UriStatus getStatus() {
057        return this.status;
058    }
059
060    /**
061     * To compare whether the two UriData is equal.
062     *
063     * @param o The object to be compared with.
064     * @return Return true if the two objects are equal, otherwise return false.
065     */
066    @Override
067    public boolean equals(Object o) {
068        if (this == o) return true;
069        if (!(o instanceof UriData)) return false;
070        UriData uriData = (UriData) o;
071        return Objects.equals(getUri(), uriData.getUri()) &&
072                getStatus() == uriData.getStatus();
073    }
074
075    /**
076     * To get the hash code of the UriData.
077     *
078     * @return Return the hash code of the UriData.
079     */
080    @Override
081    public int hashCode() {
082        return Objects.hash(getUri(), getStatus());
083    }
084
085    /**
086     * To convert the UriData into a String.
087     *
088     * @return Return the String result.
089     */
090    @Override
091    public String toString() {
092        return "UriData{" +
093                "uri='" + uri + '\'' +
094                ", status=" + status +
095                '}';
096    }
097}
