020package com.inchok.aria2;
021
022import java.util.Objects;
023
024/**
025 * The unique identifier of each download.
026 * <p>In fact, gid is a 64-bits number.</p>
027 * <p>To create a custom gid, use static method Gid.toGid(String).</p>
028 *
029 * @author inCHOK
030 * @version Version 1.0
031 */
032public class Gid {
033    private long gidNative;
034    private boolean isNewed = false;
035
036    Gid(long gidNative) {
037        this.gidNative = gidNative;
038    }
039
040    /**
041     * To get the textual representation hex of the gid.
042     *
043     * @return Return textual representation hex of the gid.
044     */
045    public String toHex() {
046        return Aria2.gidToHexNative(this.gidNative);
047    }
048
049    /**
050     * To get the gid converted from the textual representation hex.
051     *
052     * @param hexGid The textual representation hex.
053     * @return Return the gid converted from the textual representation hex.
054     */
055    public static Gid toGid(String hexGid) {
056        return new Gid(Aria2.hexToGidNative(hexGid));
057    }
058
059    /**
060     * To check whether this gid is invalid.
061     *
062     * @return Return whether this gid is invalid.
063     */
064    public boolean isNull() {
065        return Aria2.isNullNative(this.gidNative);
066    }
067
068    long getGidNative() {
069        return this.gidNative;
070    }
071
072    void setGidNative(long gidNative) {
073        this.gidNative = gidNative;
074    }
075
076    /**
077     * To get the real gid(A 64-bit number).
078     *
079     * @return Return the real gid(A 64-bit number).
080     */
081    public long getGid() {
082        return Aria2.getGidNative(this.gidNative);
083    }
084
085    /**
086     * To set the real gid(A 64-bit number).
087     *
088     * @param gid The real gid(A 64-bit number).
089     */
090    public void setGid(long gid) {
091        Aria2.setGidNative(this.gidNative, gid);
092    }
093
094    /**
095     * To compare whether the two Gids are equal.
096     *
097     * @param o The object to be compared with.
098     * @return Return true if the two objects are equal, otherwise return false.
099     */
100    @Override
101    public boolean equals(Object o) {
102        if (this == o) return true;
103        if (!(o instanceof Gid)) return false;
104        Gid gid = (Gid) o;
105        return getGidNative() == gid.getGidNative();
106    }
107
108    /**
109     * To get the hash code of the Gid.
110     *
111     * @return Return the hash code of the Gid.
112     */
113    @Override
114    public int hashCode() {
115        return Objects.hash(getGidNative());
116    }
117
118    /**
119     * To convert the Gid into a String.
120     *
121     * @return Return the String result.
122     */
123    @Override
124    public String toString() {
125        return "Gid{" +
126                "gidNative=" + gidNative +
127                '}';
128    }
129}
