020package com.inchok.aria2;
021
022/**
023 * Error code constants.
024 * <p>These error codes are returned when you call DownloadHandle.getErrorCode().</p>
025 * <p>All of these error codes are <strong>positive</strong>, some negative error codes returned by such as Aria2.initialize() <strong>have no connection</strong> with these error codes.</p>
026 *
027 * @author inCHOK
028 * @version Version 1.0
029 * @see DownloadHandle#getErrorCode()
030 */
031public class ErrorCode {
032    private ErrorCode() {
033    }
034
035    /**
036     * If an unknown error occurred.
037     */
038    public static final int UNKNOWN = 1;
039    /**
040     * If time out occurred.
041     */
042    public static final int TIME_OUT = 2;
043    /**
044     * If a resource was not found.
045     */
046    public static final int RESOURCE_NOT_FOUND = 3;
047    //TODO Option.MAX_FILE_NOT_FOUND
048    /**
049     * If aria2 saw the specified number of "resource not found" error.
050     *
051     * @see Option
052     */
053    public static final int MAX_RESOURCE_NOT_FOUND = 4;
054    //TODO Option.LOWEST_SPEED_LIMIT
055    /**
056     * If a download aborted because download speed was too slow.
057     *
058     * @see Option
059     */
060    public static final int DOWNLOAD_SPEED_TOO_SLOW = 5;
061    /**
062     * If network problem occurred.
063     */
064    public static final int NETWORK_ERROR = 6;
065    /**
066     * If there were unfinished downloads. This error is only reported if all finished downloads were successful and there were unfinished downloads in a queue when aria2 exited by pressing <cite>Ctrl-C</cite> by an user or sending <cite>TERM</cite> or <cite>INT</cite> signal.
067     */
068    public static final int KILL_BY_SIGNAL = 7;
069    /**
070     * If remote server did not support resume when resume was required to complete download.
071     */
072    public static final int RESUME_NOT_SUPPORTED = 8;
073    /**
074     * If there was not enough disk space available.
075     */
076    public static final int DISK_FULL = 9;
077    //TODO Option.ALLOW_PIECE_LENGTH_CHANGE
078    /**
079     * If piece length was different from one in .aria2 control file.
080     *
081     * @see Option
082     */
083    public static final int PIECE_LENGTH_CHANGED = 10;
084    /**
085     * If aria2 was downloading same file at that moment.
086     */
087    public static final int DOWNLOADING_SAME_FILE = 11;
088    /**
089     * If aria2 was downloading same info hash torrent at that moment.
090     */
091    public static final int DOWNLOADING_SAME_TORRENT = 12;
092    //TODO Option.ALLOW_OVERWRITING
093    /**
094     * If file already existed.
095     *
096     * @see Option
097     */
098    public static final int FILE_EXISTS = 13;
099    //TODO Option.AUTO_FILE_RENAMING
100    /**
101     * If renaming file failed.
102     *
103     * @see Option
104     */
105    public static final int FILE_RENAME_ERROR = 14;
106    /**
107     * If aria2 could not open existing file.
108     */
109    public static final int OPEN_EXISTING_FILE_ERROR = 15;
110    /**
111     * If aria2 could not create new file or truncate existing file.
112     */
113    public static final int CREATE_OR_TRUNCATE_FILE_ERROR = 16;
114    /**
115     * If file I/O error occurred.
116     */
117    public static final int FILE_IO_ERROR = 17;
118    /**
119     * If aria2 could not create directory.
120     */
121    public static final int CREATE_DIRECTORY_ERROR = 18;
122    /**
123     * If name resolution failed.
124     */
125    public static final int NAME_RESOLUTION_ERROR = 19;
126    /**
127     * If aria2 could not parse Metalink document.
128     */
129    public static final int METALINK_PARSE_ERROR = 20;
130    /**
131     * If FTP command failed.
132     */
133    public static final int FTP_COMMAND_ERROR = 21;
134    /**
135     * If HTTP response header was bad or unexpected.
136     */
137    public static final int HTTP_RESPONSE_HEADER_ERROR = 22;
138    /**
139     * If too many redirects occurred.
140     */
141    public static final int TOO_MANY_REDIRECTS = 23;
142    /**
143     * If HTTP authorization failed.
144     */
145    public static final int HTTP_AUTHORIZE_ERROR = 24;
146    /**
147     * If aria2 could not parse bencoded file (usually ".torrent" file).
148     */
149    public static final int BENCODE_PARSE_ERROR = 25;
150    /**
151     * If ".torrent" file was corrupted or missing information that aria2 needed.
152     */
153    public static final int BAD_TORRENT = 26;
154    /**
155     * If Magnet URI was bad.
156     */
157    public static final int BAD_MAGNET_URI = 27;
158    /**
159     * If bad/unrecognized option was given or unexpected option argument was given.
160     */
161    public static final int BAD_OPTION = 28;
162    /**
163     * If the remote server was unable to handle the request due to a temporary overloading or maintenance.
164     */
165    public static final int REMOTE_REFUSED = 29;
166    /**
167     * If aria2 could not parse JSON-RPC request.
168     */
169    public static final int PAUSE_RPC_ERROR = 30;
170    /**
171     * Reserved. Not used.
172     */
173    public static final int RESERVED = 31;
174    /**
175     * If checksum validation failed.
176     */
177    public static final int CHECKSUM_UNMATCH = 32;
178}
